/**
 * API Key加密工具
 * 用于安全存储敏感的API Key

import crypto from 'crypto';

// 加密密钥（应从环境变量读取）
const ENCRYPTION_KEY = process.env.PLATFORM_KEY_ENCRYPTION_KEY || '';
const ALGORITHM = 'aes-256-gcm';
const IV_LENGTH = 16;
const AUTH_TAG_LENGTH = 16;

/**
 * 加密API Key
 */
export function encryptApiKey(plainText: string): string {
  if (!ENCRYPTION_KEY || ENCRYPTION_KEY.length !== 64) {
    console.warn("[Encryption] Invalid encryption key, storing as-is (DEV MODE)");
    return plainText;
  }

  const iv = crypto.randomBytes(IV_LENGTH);
  const key = Buffer.from(ENCRYPTION_KEY, 'hex');
  
  const cipher = crypto.createCipheriv(ALGORITHM, key, iv);
  
  let encrypted = cipher.update(plainText, 'utf8', 'hex');
  encrypted += cipher.final('hex');
  
  const authTag = cipher.getAuthTag();
  
  // 返回格式: iv:authTag:encrypted
  return `${iv.toString('hex')}:${authTag.toString('hex')}:${encrypted}`;
}

/**
 * 解密API Key
 */
export function decryptApiKey(encryptedText: string): string {
  if (!ENCRYPTION_KEY || ENCRYPTION_KEY.length !== 64) {
    console.warn("[Encryption] Invalid encryption key, returning as-is (DEV MODE)");
    return encryptedText;
  }

  const parts = encryptedText.split(':');
  if (parts.length !== 3) {
    // 可能是旧版本的未加密数据
    return encryptedText;
  }

  const [ivHex, authTagHex, encrypted] = parts;
  const iv = Buffer.from(ivHex, 'hex');
  const authTag = Buffer.from(authTagHex, 'hex');
  const key = Buffer.from(ENCRYPTION_KEY, 'hex');

  const decipher = crypto.createDecipheriv(ALGORITHM, key, iv);
  decipher.setAuthTag(authTag);

  let decrypted = decipher.update(encrypted, 'hex', 'utf8');
  decrypted += decipher.final('utf8');

  return decrypted;
}

/**
 * 生成随机加密密钥（用于初始化）
 */
export function generateEncryptionKey(): string {
  return crypto.randomBytes(32).toString('hex');
}

type AlertLevel = 'info' | 'warning' | 'critical';

interface SecurityAlert {
  level: AlertLevel;
  title: string;
  message: string;
  details?: Record<string, unknown>;
  timestamp: string;
}

class AlertService {
  private alertWebhook: string | undefined = process.env.SECURITY_ALERT_WEBHOOK;
  private alertEmail: string | undefined = process.env.SECURITY_ALERT_EMAIL;

  /**
   * 发送告警
   */
  async send(alert: Omit<SecurityAlert, 'timestamp'>) {
    const fullAlert: SecurityAlert = {
      ...alert,
      timestamp: new Date().toISOString(),
    };

    // 输出到日志
    console.log(`[SECURITY_ALERT:${alert.level.toUpperCase()}]`, alert.title, alert.message);

    // 发送到Webhook（可选）
    if (this.alertWebhook) {
      await this.sendToWebhook(fullAlert);
    }

    // 发送邮件告警（可选，仅critical级别）
    if (this.alertEmail && alert.level === 'critical') {
      await this.sendEmail(fullAlert);
    }
  }

  /**
   * 管理员操作告警
   */
  async adminOperation(alert: {
    userId: string;
    action: string;
    success: boolean;
    ip: string;

        level: 'warning',
        title: '管理员操作失败',
        message: `用户 ${alert.userId} 的操作 ${alert.action} 失败`,
        details: {
          action: alert.action,
          userId: alert.userId,
          ip: alert.ip,
          error: alert.error,
        },
      });
    }
  }

  /**
   * 异常访问告警
   */
  async suspiciousAccess(details: {
    ip: string;
    userId?: string;
    endpoint: string;
    reason: string;
  }) {
    await this.send({
      level: 'critical',
      title: '可疑访问',
      message: `检测到可疑访问: ${details.reason}`,
      details,
    });
  }

  private async sendToWebhook(alert: SecurityAlert) {
    try {
      await fetch(this.alertWebhook!, {
        method: 'POST',
        headers: { 'Content-Type': 'application/json' },
        body: JSON.stringify(alert),
      });
    } catch (error) {
      console.error('[AlertService] Failed to send webhook:', error);
    }
  }

  private async sendEmail(alert: SecurityAlert) {
    // TODO: 实现邮件发送
    console.log('[AlertService] Email alert:', alert);
  }
}

export const alertService = new AlertService();

## 📋 风险分析

| 风险 | 等级 | 说明 | 缓解措施 |
|------|-----|------|---------|
| 鉴权中间件失败 | 中 | 新代码可能导致合法请求被拒绝 | 保留原有代码作为降级 |
| 数据库查询变更 | 中 | 可能影响前端数据展示 | 保持字段兼容 |
| 审计日志性能 | 低 | 写入数据库可能影响性能 | 使用异步写入 |

## 📁 需要创建的数据库表（可选）

如果需要持久化审计日志，创建以下表：

```sql
CREATE TABLE admin_audit_logs (
  id VARCHAR(36) PRIMARY KEY,
  action VARCHAR(50) NOT NULL,
  user_id VARCHAR(36) NOT NULL,
  user_role VARCHAR(20) NOT NULL,
  target VARCHAR(255) NOT NULL,
  method VARCHAR(10) NOT NULL,
  success BOOLEAN NOT NULL,
  ip VARCHAR(45),
  user_agent TEXT,
  details JSONB,
  error TEXT,
  created_at TIMESTAMP DEFAULT NOW()
);

CREATE INDEX idx_audit_user ON admin_audit_logs(user_id);
CREATE INDEX idx_audit_action ON admin_audit_logs(action);
CREATE INDEX idx_audit_created ON admin_audit_logs(created_at);
```
