## 🔧 问题4：并行执行状态竞争（严重）

### 问题描述
`handleRunAll` 使用 `Promise.allSettled` 并行执行节点，但 `inDegree` 计算方式存在竞态条件。

### 问题位置
[src/app/workspace/page.tsx:580-619](file:///d:/nvwa_projects/src/app/workspace/page.tsx#L580)

### 修复方案

```typescript
// 修改前
const handleRunAll = async () => {
  if (isRunningAll || nodes.length === 0) return;
  runAbortRef.current = false;
  setIsRunningAll(true);
  setNodes((prev) => {
    const next = prev.map((n) => ({ ...n, data: { ...n.data, status: '待开始', result: undefined } }));
    nodesRef.current = next;
    return next;
  });
  // Build in-degree map using refs to avoid closure staleness
  const inDegree = new Map<string, number>();
  nodesRef.current.forEach((n) => inDegree.set(n.id, 0));
  edgesRef.current.forEach((e) => { inDegree.set(e.target, (inDegree.get(e.target) || 0) + 1); });
  // Start with nodes that have no dependencies
  let queue = nodesRef.current.filter((n) => (inDegree.get(n.id) || 0) === 0).map((n) => n.id);
  while (queue.length > 0 && !runAbortRef.current) {
    const batch = [...queue];
    queue = [];
    await Promise.allSettled(
      batch.map(async (nodeId) => {
        if (runAbortRef.current) return;
        const ok = await executeNodeById(nodeId);
        if (runAbortRef.current) return;
        if (!ok) return;
        // Decrease in-degree of downstream nodes only on success
        const downstream = edgesRef.current.filter((e) => e.source === nodeId).map((e) => e.target);
        for (const targetId of downstream) {
          const remaining = inDegree.get(targetId) || 0;
          if (remaining > 0) {
            inDegree.set(targetId, remaining - 1);
            if (remaining - 1 === 0) { queue.push(targetId); }
          }
        }
      })
    );
  }
  setIsRunningAll(false);
  if (!runAbortRef.current) { toast.success('工作流运行完成'); }
  else { toast.info('工作流已取消'); }
};

// 修改后 - 改用串行执行确保正确性
const handleRunAll = async () => {
  if (isRunningAll || nodes.length === 0) return;
  runAbortRef.current = false;
  setIsRunningAll(true);
  setNodes((prev) => {
    const next = prev.map((n) => ({ ...n, data: { ...n.data, status: '待开始', result: undefined } }));
    nodesRef.current = next;
    return next;
  });

  // 构建拓扑排序后的执行队列
  const inDegree = new Map<string, number>();
  const adjacency = new Map<string, string[]>();
  nodesRef.current.forEach((n) => {
    inDegree.set(n.id, 0);
    adjacency.set(n.id, []);
  });
  edgesRef.current.forEach((e) => {
    inDegree.set(e.target, (inDegree.get(e.target) || 0) + 1);
    adjacency.get(e.source)?.push(e.target);
  });

  // 使用队列进行拓扑排序执行
  let queue = nodesRef.current.filter((n) => (inDegree.get(n.id) || 0) === 0).map((n) => n.id);

  while (queue.length > 0 && !runAbortRef.current) {
    const nodeId = queue.shift()!;

    if (runAbortRef.current) break;

    const ok = await executeNodeById(nodeId);

    if (runAbortRef.current) break;

    // 只有执行成功才处理下游节点
    if (ok) {
      const downstream = adjacency.get(nodeId) || [];
      for (const targetId of downstream) {
        const remaining = (inDegree.get(targetId) || 0) - 1;
        inDegree.set(targetId, remaining);
        if (remaining === 0) {
          queue.push(targetId);
        }
      }
    } else {
      // 节点执行失败：标记下游所有节点为失败
      const failedNodes = new Set<string>();
      const markFailed = (id: string) => {
        if (failedNodes.has(id)) return;
        failedNodes.add(id);
        (adjacency.get(id) || []).forEach(markFailed);
      };
      downstream.forEach(markFailed);

      if (failedNodes.size > 0) {
        setNodes((prev) => {
          const next = prev.map((n) =>
            failedNodes.has(n.id)
              ? { ...n, data: { ...n.data, status: '失败', result: '前置节点执行失败' } }
              : n
          );
          nodesRef.current = next;
          return next;
        });
      }
      break;  // 遇到失败则终止执行
    }
  }

  setIsRunningAll(false);
  if (!runAbortRef.current) { toast.success('工作流运行完成'); }
  else { toast.info('工作流已取消'); }
};
```

## 🔧 问题5：输出节点缺少错误处理（中等）

### 问题描述
输出节点没有检查上游节点是否成功执行，总是返回 true。

### 问题位置
[src/app/workspace/page.tsx:352-364](file:///d:/nvwa_projects/src/app/workspace/page.tsx#L352)

### 修复方案

```typescript
// 修改前
if (type === 'output') {
  const upstream = getUpstreamResult(nodeId);
  const content = upstream || description || '【' + label + '】\n暂无上游内容，请确保前置节点已执行。';
  setNodes((prev) => {
    const next = prev.map((n) =>
      n.id === nodeId
        ? { ...n, data: { ...n.data, status: '已完成', result: content } }
        : n
    );
    nodesRef.current = next;
    return next;
  });
  return true;
}

// 修改后
if (type === 'output') {
  const upstream = getUpstreamResult(nodeId);
  
  // 检查上游是否有有效内容
  if (!upstream) {
    const failMsg = '【' + label + '】\n暂无上游内容，请确保前置节点已执行。';
    setNodes((prev) => {
      const next = prev.map((n) =>
        n.id === nodeId
          ? { ...n, data: { ...n.data, status: '失败', result: failMsg } }
          : n
      );
      nodesRef.current = next;
      return next;
    });
    return false;
  }

  // 检查上游节点状态
  const upstreamEdge = edgesRef.current.find((e) => e.target === nodeId);
  if (upstreamEdge) {
    const upstreamNode = nodesRef.current.find((n) => n.id === upstreamEdge.source);
    const upstreamStatus = upstreamNode?.data?.status as string;
    if (upstreamStatus === '失败' || upstreamStatus === '待处理') {
      const failMsg = '【' + label + '】\n上游节点执行失败，无法输出内容。';
      setNodes((prev) => {
        const next = prev.map((n) =>
          n.id === nodeId
            ? { ...n, data: { ...n.data, status: '失败', result: failMsg } }
            : n
        );
        nodesRef.current = next;
        return next;
      });
      return false;
    }
  }

  const content = upstream || description || '【' + label + '】\n暂无上游内容。';
  setNodes((prev) => {
    const next = prev.map((n) =>
      n.id === nodeId
        ? { ...n, data: { ...n.data, status: '已完成', result: content } }
        : n
    );
    nodesRef.current = next;
    return next;
  });
  return true;
}
```
