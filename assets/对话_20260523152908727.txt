## 🔧 问题1：数据传递链路断裂（严重）

### 问题位置
[src/app/workspace/page.tsx:203-234](file:///d:/nvwa_projects/src/app/workspace/page.tsx#L203)

### 问题描述
`getUpstreamResult` 和 `getUpstreamScenes` 使用 `useCallback` 但依赖数组为空，导致返回过时数据，造成节点间数据传递断裂。

### 修复方案

```typescript
// 修改前
const getUpstreamResult = useCallback(
  (nodeId: string) => {
    const upstreamEdge = edgesRef.current.find((e) => e.target === nodeId);
    if (!upstreamEdge) return null;
    const upstreamNode = nodesRef.current.find((n) => n.id === upstreamEdge.source);
    if (!upstreamNode) return null;
    const result = upstreamNode.data?.result;
    if (result && typeof result === 'object') {
      if ('content' in result) return (result as { content: string }).content;
      if ('imageUrl' in result) return (result as { imageUrl: string }).imageUrl;
      if ('videoUrl' in result) return (result as { videoUrl: string }).videoUrl;
    }
    if (typeof result === 'string') return result;
    return (upstreamNode.data?.description as string) || null;
  },
  []
);

const getUpstreamScenes = useCallback(
  (nodeId: string) => {
    const upstreamEdge = edgesRef.current.find((e) => e.target === nodeId);
    if (!upstreamEdge) return [] as Array<Record<string, unknown>>;
    const upstreamNode = nodesRef.current.find((n) => n.id === upstreamEdge.source);
    if (!upstreamNode) return [] as Array<Record<string, unknown>>;
    const result = upstreamNode.data?.result;
    if (result && typeof result === 'object' && Array.isArray((result as Record<string, unknown>).scenes)) {
      return (result as Record<string, unknown>).scenes as Array<Record<string, unknown>>;
    }
    return [] as Array<Record<string, unknown>>;
  },
  []
);

// 修改后 - 移除 useCallback，直接定义函数
function getUpstreamResult(nodeId: string): string | null {
  const upstreamEdge = edgesRef.current.find((e) => e.target === nodeId);
  if (!upstreamEdge) return null;
  const upstreamNode = nodesRef.current.find((n) => n.id === upstreamEdge.source);
  if (!upstreamNode) return null;
  const result = upstreamNode.data?.result;
  if (result && typeof result === 'object') {
    if ('content' in result) return (result as { content: string }).content;
    if ('imageUrl' in result) return (result as { imageUrl: string }).imageUrl;
    if ('videoUrl' in result) return (result as { videoUrl: string }).videoUrl;
  }
  if (typeof result === 'string') return result;
  return (upstreamNode.data?.description as string) || null;
}

function getUpstreamScenes(nodeId: string): Array<Record<string, unknown>> {
  const upstreamEdge = edgesRef.current.find((e) => e.target === nodeId);
  if (!upstreamEdge) return [];
  const upstreamNode = nodesRef.current.find((n) => n.id === upstreamEdge.source);
  if (!upstreamNode) return [];
  const result = upstreamNode.data?.result;
  if (result && typeof result === 'object' && Array.isArray((result as Record<string, unknown>).scenes)) {
    return (result as Record<string, unknown>).scenes as Array<Record<string, unknown>>;
  }
  return [];
}
```

## 🔧 问题5：输出节点缺少上游失败处理（中等）

### 问题位置
[src/app/workspace/page.tsx:352-364](file:///d:/nvwa_projects/src/app/workspace/page.tsx#L352)

### 问题描述
输出节点没有检查上游节点的执行状态，即使上游失败仍标记为"已完成"。

### 修复方案

```typescript
// 修改前
if (type === 'output') {
  const upstream = getUpstreamResult(nodeId);
  const content = upstream || description || '【' + label + '】\n暂无上游内容，请确保前置节点已执行。';
  setNodes((prev) => {
    const next = prev.map((n) =>
      n.id === nodeId
        ? { ...n, data: { ...n.data, status: '已完成', result: content } }
        : n
    );
    nodesRef.current = next;
    return next;
  });
  return true;
}

// 修改后
if (type === 'output') {
  const upstream = getUpstreamResult(nodeId);
  
  // 检查上游是否有有效内容
  if (!upstream) {
    const failMsg = '【' + label + '】\n暂无上游内容，请确保前置节点已执行。';
    setNodes((prev) => {
      const next = prev.map((n) =>
        n.id === nodeId
          ? { ...n, data: { ...n.data, status: '失败', result: failMsg } }
          : n
      );
      nodesRef.current = next;
      return next;
    });
    return false;
  }

  // 检查上游节点状态
  const upstreamEdge = edgesRef.current.find((e) => e.target === nodeId);
  if (upstreamEdge) {
    const upstreamNode = nodesRef.current.find((n) => n.id === upstreamEdge.source);
    const upstreamStatus = upstreamNode?.data?.status as string;
    if (upstreamStatus === '失败' || upstreamStatus === '待处理') {
      const failMsg = '【' + label + '】\n上游节点执行失败，无法输出内容。';
      setNodes((prev) => {
        const next = prev.map((n) =>
          n.id === nodeId
            ? { ...n, data: { ...n.data, status: '失败', result: failMsg } }
            : n
        );
        nodesRef.current = next;
        return next;
      });
      return false;
    }
  }

  const content = upstream || description || '【' + label + '】\n暂无上游内容。';
  setNodes((prev) => {
    const next = prev.map((n) =>
      n.id === nodeId
        ? { ...n, data: { ...n.data, status: '已完成', result: content } }
        : n
    );
    nodesRef.current = next;
    return next;
  });
  return true;
}
```
