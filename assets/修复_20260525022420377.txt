```typescript
/**
 * Redis队列实现
 * 生产环境必须使用Redis替代内存Map
 */

import Redis from 'ioredis';

const redis = new Redis(process.env.REDIS_URL || 'redis://localhost:6379');

export interface QueueJob<T = unknown> {
  id: string;
  data: T;
  attempts: number;
  maxAttempts: number;
  status: 'pending' | 'processing' | 'completed' | 'failed';
  createdAt: string;
  processAt?: string;
}

/**
 * 添加任务到队列
 */
export async function addJob<T>(
  queueName: string,
  data: T,
  options: { maxAttempts?: number; delay?: number } = {}
): Promise<string> {
  const jobId = `job_${Date.now()}_${Math.random().toString(36).slice(2)}`;
  
  const job: QueueJob<T> = {
    id: jobId,
    data,
    attempts: 0,
    maxAttempts: options.maxAttempts || 3,
    status: 'pending',
    createdAt: new Date().toISOString(),
    processAt: options.delay 
      ? new Date(Date.now() + options.delay).toISOString() 
      : undefined,
  };

  await redis.lpush(`${queueName}:queue`, JSON.stringify(job));
  
  return jobId;
}

/**
 * 获取待处理任务
 */
export async function getJob<T>(queueName: string): Promise<QueueJob<T> | null> {
  const job = await redis.rpop(`${queueName}:queue`);
  return job ? JSON.parse(job) : null;
}

/**
 * 完成处理任务
 */
export async function completeJob(queueName: string, jobId: string): Promise<void> {
  await redis.lpush(`${queueName}:completed`, jobId);
}

/**
 * 失败任务处理
 */
export async function failJob(queueName: string, job: QueueJob): Promise<void> {
  if (job.attempts < job.maxAttempts) {
    // 重试
    job.attempts++;
    job.status = 'pending';
    job.processAt = new Date(Date.now() + 5000).toISOString(); // 5秒后重试
    await redis.lpush(`${queueName}:queue`, JSON.stringify(job));
  } else {
    // 放弃
    await redis.lpush(`${queueName}:failed`, JSON.stringify(job));
  }
}
```

#### 4.2.2 图片压缩实现

```typescript
/**
 * 图片压缩工具
 * 使用Sharp进行服务端图片压缩
 */

import sharp from 'sharp';

export interface CompressOptions {
  quality?: number; // 1-100
  maxWidth?: number;
  maxHeight?: number;
  format?: 'jpeg' | 'png' | 'webp';
}

/**
 * 压缩图片
 */
export async function compressImage(
  buffer: Buffer,
  options: CompressOptions = {}
): Promise<Buffer> {
  const {
    quality = 80,
    maxWidth = 1920,
    maxHeight = 1920,
    format = 'jpeg',
  } = options;

  let transformer = sharp(buffer)
    .resize(maxWidth, maxHeight, {
      fit: 'inside',
      withoutEnlargement: true,
    });

  switch (format) {
    case 'jpeg':
      transformer = transformer.jpeg({ quality });
      break;
    case 'png':
      transformer = transformer.png({ quality });
      break;
    case 'webp':
      transformer = transformer.webp({ quality });
      break;
  }

  return transformer.toBuffer();
}

/**
 * 生成缩略图
 */
export async function generateThumbnail(
  buffer: Buffer,
  size: number = 200
): Promise<Buffer> {
  return sharp(buffer)
    .resize(size, size, { fit: 'cover' })
    .jpeg({ quality: 70 })
    .toBuffer();
}
```

```typescript
/**
 * 用户行为埋点系统
 * 支持多种分析工具集成
 */

export interface TrackEvent {
  event: string;
  properties?: Record<string, unknown>;
  timestamp?: string;
}

class Analytics {
  private endpoint: string | null = null;

  /**
   * 初始化埋点
   */
  init(config: { endpoint?: string; debug?: boolean }) {
    this.endpoint = config.endpoint || null;
    
    if (config.debug) {
      console.log('[Analytics] Initialized with endpoint:', this.endpoint);
    }
  }

  /**
   * 追踪页面访问
   */
  trackPage(page: string, properties?: Record<string, unknown>) {
    this.track('page_view', {
      page,
      url: typeof window !== 'undefined' ? window.location.href : '',
      referrer: typeof document !== 'undefined' ? document.referrer : '',
      ...properties,
    });
  }

  /**
   * 追踪用户行为
   */
  track(event: string, properties?: Record<string, unknown>) {
    const payload: TrackEvent = {
      event,
      properties,
      timestamp: new Date().toISOString(),
    };

    // 开发环境打印
    if (process.env.NODE_ENV === 'development') {
      console.log('[Analytics]', payload);
    }

    // 发送到服务器
    if (this.endpoint) {
      this.sendToServer(payload);
    }
  }

  private async sendToServer(payload: TrackEvent) {
    try {
      await fetch(`${this.endpoint}/api/analytics`, {
        method: 'POST',
        headers: { 'Content-Type': 'application/json' },
        body: JSON.stringify(payload),
      });
    } catch (error) {
      console.error('[Analytics] Failed to send:', error);
    }
  }
}

export const analytics = new Analytics();

// 自动追踪页面访问
if (typeof window !== 'undefined') {
  window.addEventListener('popstate', () => {
    analytics.trackPage(window.location.pathname);
  });
}
```

**创建Hook：** `/src/hooks/useMobileGestures.ts`

```typescript
"use client";

import { useState, useEffect, useCallback } from "react";

/**
 * 移动端手势检测
 */
export function useMobileGestures(elementRef: React.RefObject<HTMLElement>) {
  const [swipeDirection, setSwipeDirection] = useState<string | null>(null);
  const [pinchScale, setPinchScale] = useState(1);

  useEffect(() => {
    const element = elementRef.current;
    if (!element) return;

    let startX = 0;
    let startY = 0;
    let initialDistance = 0;

    const handleTouchStart = (e: TouchEvent) => {
      if (e.touches.length === 1) {
        startX = e.touches[0].clientX;
        startY = e.touches[0].clientY;
      } else if (e.touches.length === 2) {
        initialDistance = Math.hypot(
          e.touches[0].clientX - e.touches[1].clientX,
          e.touches[0].clientY - e.touches[1].clientY
        );
      }
    };

    const handleTouchMove = (e: TouchEvent) => {
      if (e.touches.length === 2) {
        const currentDistance = Math.hypot(
          e.touches[0].clientX - e.touches[1].clientX,
          e.touches[0].clientY - e.touches[1].clientY
        );
        setPinchScale(currentDistance / initialDistance);
      }
    };

    const handleTouchEnd = (e: TouchEvent) => {
      const endX = e.changedTouches[0].clientX;
      const endY = e.changedTouches[0].clientY;
      const diffX = endX - startX;
      const diffY = endY - startY;

      if (Math.abs(diffX) > Math.abs(diffY)) {
        if (Math.abs(diffX) > 50) {
          setSwipeDirection(diffX > 0 ? 'right' : 'left');
        }
      } else {
        if (Math.abs(diffY) > 50) {
          setSwipeDirection(diffY > 0 ? 'down' : 'up');
        }
      }
    };

    element.addEventListener('touchstart', handleTouchStart);
    element.addEventListener('touchmove', handleTouchMove);
    element.addEventListener('touchend', handleTouchEnd);

    return () => {
      element.removeEventListener('touchstart', handleTouchStart);
      element.removeEventListener('touchmove', handleTouchMove);
      element.removeEventListener('touchend', handleTouchEnd);
    };
  }, [elementRef]);

  return { swipeDirection, pinchScale };
}
```
